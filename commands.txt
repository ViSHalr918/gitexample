git congif --global user.name "username"        [to add your username]

git congif --global user.email "gmail"        [to add your gmail]


git init                                      [to initialize an empty git repository]


git status                                    [to check untracked files]

git add -A                                    [to add all untracked files to staging area]


git commit -m "commit message"                 [to move staged files to local repository]

git remote add remote_name remote_path        [to add remote repository]


git push remote_name branch                   [push to remote repository]


git branch                                    [to list all branches]


git remote -v                                 [to list all remote repository]
